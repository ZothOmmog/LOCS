--Изменение мероприятия
CREATE OR REPLACE PROCEDURE ChangeEvent(idev_ bigint, name_ text,info_ text,
										link_ text, ticket_price_ numeric, id_org bigint,
										id_address_ bigint, publish_ bool default false)
    LANGUAGE 'plpgsql'
AS $BODY$
BEGIN 
	update event set name = name_, info = info_, link = link_, ticket_price = ticket_price_,id_organizer = id_org,
					id_address = id_address_, published = publish_ where id = idev_; 
END;
$BODY$;
