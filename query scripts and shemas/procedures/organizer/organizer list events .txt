--данные об мероприятиях для организатора
CREATE OR REPLACE FUNCTION organizerEvents(id_ bigint)
RETURNS table (j json) AS
$$
BEGIN
    RETURN QUERY  SELECT to_json(a) FROM (SELECT event.id, event.name, event.info, event.link, event.ticket_price,
										  event.id_address,event.image,event.published from event 
										  where event.id = id_ and event.deleted = false) a ;
END;
$$ LANGUAGE plpgsql;
